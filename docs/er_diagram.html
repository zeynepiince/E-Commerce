    users {
        int user_id PK
        varchar full_name
        varchar email UK
        varchar password_hash
        varchar google_id
        varchar facebook_id
        varchar phone
        text address
        tinyint email_notifications
        timestamp created_at
    }

    categories {
        int category_id PK
        varchar category_name UK
    }

    products {
        int product_id PK
        varchar external_id UK
        varchar name
        decimal price
        int category_id FK
        int stock_quantity
        json badges
    }

    orders {
        int order_id PK
        int user_id FK
        decimal total_amount
        varchar status
        varchar payment_status
        text shipping_json
        timestamp created_at
    }

    order_items {
        int order_item_id PK
        int order_id FK
        int product_id FK
        int quantity
        decimal unit_price
    }

    payments {
        int payment_id PK
        int order_id
        varchar provider
        varchar token
        varchar status
        decimal amount
    }

    password_resets {
        int reset_id PK
        int user_id FK
        char token_hash
        datetime expires_at
    }

    support_interactions {
        int interaction_id PK
        int user_id FK
        text message
        enum sender
        int recommended_product_id FK
    }

    chatbot_feedback {
        int feedback_id PK
        int user_id FK
        tinyint helpful
        varchar intent
        tinyint used_ai
    }
